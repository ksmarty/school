import { saveAs } from "file-saver";

export const getPDF = async () => {
	const paths = window.location.pathname
		.split("/")
		.filter((e) => e.length > 0)
		.map((e) => decodeURI(e));
	// console.log(`${paths[0]} - ${paths[1]}.pdf`);
	let response = await fetch(
		"https://cdn.discordapp.com/attachments/799412047501590528/876144049067806790/Getting_Started.pdf"
	);
	const data = await response.blob();
	const blob = new Blob(data, { type: "application/pdf" });
	saveAs(blob, `${paths[0]} - ${paths[1]}.pdf`);
};
